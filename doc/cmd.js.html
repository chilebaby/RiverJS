var _$river = {
  // module define and run api
  sandbox: function() {
    var boxes = {};
    return {
      create: function(key, fn) {
        key = key.toLowerCase();
        boxes[key] = fn;
      },
      run: function(fn) {
        var context = {
          need: function(key) {
            key = key.toLowerCase();
            return boxes[key] && boxes[key].call(context) || undefined;
          }
        };
        fn.call(context);
      }
    };
  }
};
_$river.module = _$river.sandbox();
